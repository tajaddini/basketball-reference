Table players {
  player_id varchar unique pk
  name varchar
  pos varchar
  shoots varchar
  age float
  is_alive bool
  height float
  weight float
  career_length int
  is_active bool
  has_hall_of_fame bool
  count_allstar int
  stat_games int
  stat_points int
  stat_total_rebounds int
  stat_assists int
  stat_field_goal_pct float
  stat_three_point_field_goal_pct float
  stat_effective_field_goal_pct float
  stat_free_throw_pct float
  stat_efficiency_rating float
  stat_win_share float
}

Table kpis {
  id int pk
  player_id varchar
  season varchar
  kpi_1 float
  kpi_2 float
}

Table salaries {
  id int [primary key]
  player_id varchar
  season varchar
  salary int
}

Table roster_data {
  id int [primary key]
  player_id varchar
  team_name varchar
  season varchar
  player_position varchar
}
